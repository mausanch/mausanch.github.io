Ejercicio 1

a)Analítico

figure
subplot(1,3,1)
syms t
x=piecewise(t<0,0,0<t<1,-t+1,1<t<2,t-1,t>2,0);
fplot(x,[-1,3],'r')
grid on
title('x(t)')
axis([-1 3.6 -2 2]);
subplot(1,3,2)
h=piecewise(t<0,0,0<t<1,1,t>1,0);
fplot(h,[-1,3],'r')
grid on
title('h(t)')
axis([-1 3.6 -2 2]);
t1=0:0.01:1;
t2=1:0.01:2;
t3=2:0.01:3;
subplot(1,3,3)
plot(t1,(((-t1.^2)/2)+t1));
hold on
plot(t2,(t2.^2)-(3.*t2)+(5/2));
plot(t3,((-t3.^2)/2)+(2.*t3)-(3/2));
grid on
title('x(t)*h(t)')
axis([-1 4 -0.1 0.6]);

b)Calculado por MATLAB
x=@(t)(-t+1).*(t>=0 & t<1)+(t-1).*(t>=1 & t<2);
h=@(t)(1).*(t>=0 & t<1);
convconm(x,h);
