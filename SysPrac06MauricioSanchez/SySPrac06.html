hFig = figure(1);
set(hFig, 'Position', [0 0 900 900])
axes1 = axes('Parent',hFig,'FontWeight','bold','FontSize',16);
t0=10;
f=dirac(t);
tiempo=0:t0/100:t0;
subplot(3,2,1)
y=laplace2016woplots([2,2,1],[2,1],[0,0],[0],f,10);
fplot(matlabFunction(f),[0, t0],'b')

hold on
fplot(y,[0,t0],'r')
hold on

legend('Entrada x(t)','Salida y(t)','Location','Best')
title('Respuesta al impulso','FontWeight','bold','FontSize',16)
grid on
